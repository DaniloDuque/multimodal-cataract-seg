% ── Slide 3: ¿Qué es U-Net? ─────────────────────────────────────────────────
\begin{frame}{¿Qué es U-Net?}
\begin{columns}[T]
\column{0.58\textwidth}
Arquitectura \textbf{encoder-decoder} para segmentación:
\begin{itemize}
    \item \textbf{Encoder}: reduce resolución, aumenta canales. Captura \textbf{qué} hay.
    \item \textbf{Decoder}: aumenta resolución, reduce canales. Recupera \textbf{dónde}.
    \item \textbf{Skip connections}: conectan encoder y decoder. \textbf{Preservan el detalle fino} que se pierde al submuestrear.
\end{itemize}

\column{0.38\textwidth}
\centering
\begin{tikzpicture}[font=\tiny, scale=0.55, transform shape]
    \fill[blue!15, rounded corners=3pt] (0,0) rectangle (1.5,3);
    \fill[red!15, rounded corners=3pt] (3,0) rectangle (4.5,3);
    \fill[gray!15, rounded corners=3pt] (1.5,1.2) rectangle (3,1.8);
    \node[rotate=90] at (0.75,1.5) {\textbf{Encoder}};
    \node[rotate=-90] at (3.75,1.5) {\textbf{Decoder}};
    \node at (2.25,1.5) {\textbf{Botella}};
    \draw[blue!60, dashed, thick] (1.5,2.8) -- (3,2.8);
    \draw[blue!60, dashed, thick] (1.5,2.0) -- (3,2.0);
    \draw[blue!60, dashed, thick] (1.5,0.8) -- (3,0.8);
    \draw[-{Stealth}, thick] (0,0) -- (0,3);
    \draw[-{Stealth}, thick] (1.5,1.5) -- (3,1.5);

\end{tikzpicture}
\end{columns}
\begin{block}{Clave}
Usamos U-Net como backbone porque sus skip connections preservan los bordes anatómicos finos para segmentar córnea y cristalino.
\end{block}
\end{frame}

% ── Slide 4: ¿Qué es Atención? ──────────────────────────────────────────────
\begin{frame}{¿Qué es Atención (Attention)?}
\begin{columns}[T]
\column{0.55\textwidth}
Mecanismo que permite a un modelo \textbf{enfocarse} en las partes relevantes.

\textbf{Tres componentes:}
\begin{itemize}
    \item \textbf{Q (Query)}: ``¿qué información necesito?''
    \item \textbf{K (Key)}: ``¿qué información tengo?''
    \item \textbf{V (Value)}: ``el contenido que voy a entregar''
\end{itemize}
\vspace{0.3em}
\textbf{Cálculo:} se compara Q con cada K, se obtiene un peso de relevancia y se combinan los V.

\column{0.42\textwidth}
\begin{equation*}
\text{Atención}(Q,K,V) = \text{softmax}\!\left(\frac{QK^\top}{\sqrt{d}}\right)V
\end{equation*}
\end{columns}
\vspace{0.3em}
\begin{block}{Analogía}
Como buscar en Google: tu \textbf{consulta} (Q) se compara con \textbf{claves} (K) de páginas web y recibes los \textbf{contenidos} (V) más relevantes.
\end{block}
\end{frame}

% ── Slide 6: ¿Qué es Canny? ─────────────────────────────────────────────────
\begin{frame}{¿Qué es el detector de bordes Canny?}
Transforma una imagen a color en un \textbf{mapa de bordes}.
\begin{enumerate}
    \item \textbf{Suavizar}: filtro Gaussiano para eliminar ruido.
    \item \textbf{Gradiente}: en cada píxel, qué tan rápido cambia el brillo.
    \item \textbf{Adelgazar}: solo los máximos locales en la dirección del gradiente.
    \item \textbf{Conectar bordes} con dos umbrales ($t_1=50$, $t_2=150$):
    \begin{itemize}
        \item $> t_2$: borde fuerte (se queda)
        \item $< t_1$: se descarta
        \item Entre $t_1$ y $t_2$: solo si conecta a un borde fuerte
    \end{itemize}
\end{enumerate}
\vspace{0.3em}
{\small Dos umbrales producen bordes \textbf{limpios y conectados}: eliminan ruido pero mantienen contornos relevantes.}
\end{frame}

% ── Slide 7: ¿Qué es IoU y Dice? ─────────────────────────────────────────────
\begin{frame}{Métricas: IoU y Dice}
\begin{columns}[T]
\column{0.48\textwidth}
\centering
\textbf{IoU (Intersection over Union)}
\begin{equation*}
\text{IoU} = \frac{|\hat{M} \cap M|}{|\hat{M} \cup M|}
\end{equation*}
{\small Solapamiento / área total. 1 = perfecto, 0 = nada.}

\includegraphics[width=0.55\textwidth]{../figures/iou_comparison.pdf}

\column{0.48\textwidth}
\centering
\textbf{Dice}
\begin{equation*}
\text{Dice} = \frac{2|\hat{M} \cap M|}{|\hat{M}| + |M|}
\end{equation*}
{\small Pesa más el solapamiento. Equivalente a F1.}
\vspace{0.5em}

\begin{block}{Relación}
IoU es más \textbf{conservador}: $\text{IoU} \leq \text{Dice}$. Si IoU=0.95, Dice$\approx$0.974.
Ambos penalizan FP y FN.
\end{block}
\end{columns}
\end{frame}

% ── Slide 10: ¿Qué es segmentación multimodal? ────────────────────────────────
\begin{frame}{¿Qué es Segmentación Multimodal?}
\begin{block}{Idea central}
Usar \textbf{múltiples representaciones} de la misma escena y \textbf{fusionarlas} para que el modelo vea más de lo que cualquier vista individual puede capturar.
\end{block}
\vspace{0.6em}
\begin{columns}
\column{0.48\textwidth}
\textbf{Multimodal clásico} (sensores distintos):
\begin{itemize}
    \item MRI T1 + T2 + FLAIR para tumores cerebrales
    \item RGB + profundidad para escenas 3D
\end{itemize}
\column{0.48\textwidth}
\textbf{Nuestro enfoque} (misma imagen, dos vistas):
\begin{itemize}
    \item Imagen RGB $\rightarrow$ features fotométricas
    \item Mapa de bordes Canny $\rightarrow$ features geométricas
    \item Espacios de features heterogéneos $\Rightarrow$ fusión multimodal
\end{itemize}
\end{columns}
\end{frame}

% ── Slide 15: Flujo completo del modelo ──────────────────────────────────────
\begin{frame}{Flujo Completo del Modelo}
\footnotesize
\textbf{1. Entrada.} Una imagen RGB $512^2$ alimenta el \textcolor{blue!80}{Encoder RGB}. La misma pasa por Canny ($t_1{=}50$, $t_2{=}150$) y el mapa de bordes alimenta el \textcolor{red!80}{Encoder de bordes}. Ambos son ResNet-34 sin pesos compartidos.

\medskip
\textbf{2. Encoders.} Cada uno reduce resolución y aumenta canales en 5 etapas: $64{\to}64{\to}128{\to}256{\to}512$ canales, desde $256^2$ hasta $16^2$ en el cuello de botella.

\medskip
\textbf{3. Cross-Attention.} En stage 4 ($32^2$, 256ch) y bottleneck ($16^2$, 512ch): RGB $\to$ \textbf{Q}, bordes $\to$ \textbf{K,V}. Cada uno tiene un gate $\alpha{=}\sigma(\theta)$ con $\theta{=}{-}4$, para que el modelo aprenda a usar bordes gradualmente.

\medskip
\textbf{4. Decoder.} Parte del bottleneck fusionado y expande: $512{\to}256{\to}128{\to}64$ canales hasta la máscara $512^2$.

\medskip
\textbf{5. Skip connections duales.} En cada nivel del decoder, la feature RGB y la de bordes se concatenan por canales y pasan por una Conv $1\times1$ que proyecta al canal original.

\medskip
\textbf{6. Salida.} Máscara binaria $512^2$ con sigmoid.
\end{frame}

% ── Slide 18: Encoders duales ─────────────────────────────────────────────────
\begin{frame}{Encoders Duales}
\begin{itemize}
    \item Dos encoders U-Net \textbf{independientes} — misma arquitectura ResNet-34, \textbf{sin pesos compartidos}
    \vspace{0.3em}
    \item Cada encoder produce mapas de features en \textbf{5 resoluciones}:
    \begin{itemize}
        \item Canales: 64 → 64 → 128 → 256 → 512
        \item Resolución espacial: 256 → 128 → 64 → 32 → 16
    \end{itemize}
    \vspace{0.3em}
    \item \textbf{Encoder A} procesa la imagen RGB
    \item \textbf{Encoder B} procesa el mapa de bordes
\end{itemize}
\vspace{0.3em}
\begin{block}{Clave}
Los pesos separados permiten que cada encoder aprenda representaciones especializadas para su modalidad.
\end{block}
\end{frame}

% ── Slide 19: Módulo de Cross-Attention ──────────────────────────────────────
\begin{frame}{Módulo de Cross-Attention}
\begin{columns}
\column{0.55\textwidth}
Reshape: $(B, C, H, W) \rightarrow (B,\, HW,\, C)$

\vspace{0.4em}
\begin{equation*}
\tilde{\mathbf{F}}_A = \text{LN}\!\left(\alpha \cdot \text{MHA}(\underbrace{Q{=}\mathbf{F}_A}_{\text{RGB}},\; \underbrace{K{=}V{=}\mathbf{F}_B}_{\text{bordes}}) + \mathbf{F}_A\right)
\end{equation*}

\vspace{0.3em}
\begin{itemize}
    \item \textbf{Q} viene del encoder RGB
    \item \textbf{K, V} vienen del encoder de bordes
    \item El encoder RGB \textbf{consulta} los features de bordes
    \item 8 cabezas de atención
\end{itemize}

\column{0.42\textwidth}
\begin{block}{¿Qué logra esto?}
Enriquece la representación fotométrica con información estructural geométrica antes del decodificador.
\end{block}
\end{columns}
\end{frame}

% ── Slide 20: Gate aprendible ─────────────────────────────────────────────────
\begin{frame}{Gate Escalar Aprendible}
\begin{columns}
\column{0.55\textwidth}
\begin{equation*}
\alpha = \sigma(\theta), \quad \theta_0 = -4 \Rightarrow \alpha_0 \approx 0.02
\end{equation*}
\vspace{0.3em}
\begin{itemize}
    \item Al inicio: modelo casi puramente RGB
    \item Durante el entrenamiento: $\alpha$ crece \textbf{solo si los bordes ayudan}
    \item Si el mapa Canny es ruidoso: el gate permanece cerrado
\end{itemize}

\column{0.42\textwidth}
\begin{alertblock}{¿Por qué es importante?}
Sin el gate, los features de bordes ruidosos degradarían el rendimiento. El gate protege al modelo.
\end{alertblock}
\vspace{0.3em}
\begin{block}{Diferencia con Early Fusion}
Early Fusion fuerza ambas modalidades desde la época 1. El gate es selectivo.
\end{block}
\end{columns}
\end{frame}

% ── Slide 21: Fusión multi-escala ─────────────────────────────────────────────
\begin{frame}{Fusión Multi-Escala}
\textbf{Dos instancias de cross-attention:}
\vspace{0.3em}
\begin{enumerate}
    \item \textbf{Cuello de botella} — 512 canales, resolución $16{\times}16$
    \item \textbf{Etapa 4} — 256 canales, resolución $32{\times}32$
\end{enumerate}
\vspace{0.5em}
\textbf{Skip connections del encoder de bordes:}
\begin{equation*}
\hat{\mathbf{F}}^{(s)} = \text{Conv}_{1\times1}\!\left(\left[\mathbf{F}_A^{(s)} \;\|\; \mathbf{F}_B^{(s)}\right]\right), \quad s \in \{1,2,3,4\}
\end{equation*}
\vspace{0.2em}
\begin{block}{Efecto combinado}
Los cues geométricos se propagan por \textbf{todo el camino de expansión}, no solo en el cuello de botella.
\end{block}
\end{frame}

% ── Slide 24: Métricas ────────────────────────────────────────────────────────
\begin{frame}{Métricas de Evaluación}
\begin{itemize}
    \item \textbf{IoU (Intersection over Union)} — métrica primaria
    \begin{equation*}
        \text{IoU} = \frac{|\hat{M} \cap M|}{|\hat{M} \cup M|}
    \end{equation*}
    Penaliza tanto falsos positivos como falsos negativos
    \vspace{0.4em}
    \item \textbf{Dice / F1} — métrica secundaria
    \begin{equation*}
        \text{Dice} = \frac{2|\hat{M} \cap M|}{|\hat{M}| + |M|}
    \end{equation*}
    Equivalente a F1 en segmentación binaria
\end{itemize}
\end{frame}

\hline
U-Net (RGB)            & 0.9529 & 0.9759 \\
U-Net (Bordes)         & 0.8817 & 0.9371 \\
U-Net (Fusión T.)      & \textbf{0.9532} & \textbf{0.9761} \\
\textbf{Propuesto}     & 0.9530 & 0.9759 \\
\hline
\end{tabular}
\vspace{0.5em}
\begin{itemize}
    \item Los bordes solos confirman que Canny no es suficiente de forma independiente (IoU 0.88)
    \item Los 3 modelos basados en RGB están dentro de \textbf{0.0003 IoU} — ruido estadístico a $n=45$
    \item El modelo propuesto \textbf{empata en primer lugar} con un mecanismo más principiado
\end{itemize}
\end{frame}

% ── Slide 31: Lo que hicimos ──────────────────────────────────────────────────
\begin{frame}{Conclusión: Lo que Hicimos}
\begin{itemize}
    \item Encoder dual U-Net: RGB + mapa de bordes Canny
    \vspace{0.4em}
    \item Cross-attention con gate aprendible en cuello de botella + etapa 4
    \vspace{0.4em}
    \item Skip connections del encoder de bordes en todos los niveles del decodificador
\end{itemize}
\vspace{0.5em}
\begin{alertblock}{Limitaciones}
\begin{itemize}
    \item Test set de 45 imágenes — diferencias dentro del ruido estadístico
    \item Umbrales Canny fijos
    \item Sin tabla de ablación
\end{itemize}
\end{alertblock}
\end{frame}
